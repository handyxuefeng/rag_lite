# 项目启动后，会因为终端断开而导致服务停止运行
 - 解决方案：使用 systemd（Linux 系统标准的服务管理工具）来管理你的 Flask 服务
 步骤 1：创建服务配置文件
首先创建一个 systemd 服务文件，建议放在 /etc/systemd/system/ 目录下（需要 root 权限）：
# 编辑服务文件（替换 flask-app 为你自己的服务名，比如项目名）
vim /etc/systemd/system/flask-rag-lite.service



**将以下内容粘贴进去，根据你的实际路径修改关键配置,记得把 # 去掉 **
```
[Unit]
Description=Flask Rag Lite Service  # 知识库服务
After=network.target  # 网络启动后再启动该服务

[Service]
# 运行用户（建议用非 root 用户，比如 www-data 或你自己的用户，避免权限过大）
User=root
Group=root

# 项目工作目录（替换为你的 main.py 所在的绝对路径）
WorkingDirectory=/root/rag_lite

# 启动命令（完全复制你手动启动的命令，确保路径准确）
ExecStart=/root/.local/bin/uv run --python /usr/local/python3.12/bin/python3.12 main.py

# 进程退出后的重启策略（异常退出/崩溃时自动重启）
Restart=always
RestartSec=5  # 重启间隔 5 秒

# 日志输出配置（方便排查问题）
StandardOutput=journal+console
StandardError=journal+console

# 环境变量（如果你的项目需要，比如 FLASK_ENV、数据库连接等）
Environment="PATH=/root/.local/bin:/usr/local/python3.12/bin:/usr/bin:/bin"

[Install]
WantedBy=multi-user.target  # 开机自启的目标级别

```



# 步骤 2：重新加载 systemd 配置并启动服务
# 重新加载 systemd 配置（修改服务文件后必须执行）
systemctl daemon-reload

# 启动服务
systemctl start flask-rag-lite

# 设置开机自启
systemctl enable flask-rag-lite

# 查看服务状态（确认是否正常运行）
systemctl status flask-rag-lite


# 步骤 3：常用运维命令

# 重启服务（修改代码后）
systemctl restart flask-rag-lite

# 停止服务
systemctl stop flask-rag-lite

# 查看服务日志（排查问题）
journalctl -u flask-rag-lite -f  # -f 实时查看日志
journalctl -u flask-rag-lite --since "10 minutes ago"  # 查看最近10分钟日志
